Module Interface

class BaseModule:
    def __init__(self, module_path: str):
        self.name = module_name
        self.config = load_config(module_path + "/config.json")
        self.payloads = load_payloads(module_path + "/payloads.txt")
        self.passive_scanner = PassiveScanner()  # R7: Passive analysis

    def scan(self, targets: List[Dict], http_client) -> List[Dict]:
        """Main scan method - must be implemented by each module"""
        results = []
        for target in targets:
            for payload in self.payloads:
                response = http_client.get(target['url'], params={'param': payload})

                # R7: Analyze payload response with passive scanner!
                self.analyze_payload_response(response, target['url'], payload)

                if self.detect_vulnerability(response, payload):
                    results.append(self.create_result(...))
        return results

    def create_result(self, vulnerable, url, parameter, payload, evidence, confidence):
        """Standardized result format"""
        return {
            'vulnerable': vulnerable,
            'module': self.name,
            'url': url,
            'parameter': parameter,
            'payload': payload,
            'evidence': evidence,
            'confidence': confidence,
            'severity': self.config.get('severity'),
            'cwe': self.config.get('cwe'),
            'owasp': self.config.get('owasp')
        }
